DECLARE
v_days NUMBER; v_fine NUMBER := 0; v_roll NUMBER := &roll_no; v_book VARCHAR2(50) := '&book_name';

v_days := &days_since_issue;
IF v_days BETWEEN 15 AND 30 THEN v_fine := v_days * 5;

INSERT INTO N_RollCall VALUES (1, 'Alice', 'Present'); INSERT INTO N_RollCall VALUES (2, 'Bob', 'Absent');

3. PL/SQL CODE BLOCK (TYPE / AND HIT ENTER TO RUN)
DECLARE
CURSOR c_merge IS SELECT * FROM N_RollCall; v_exists NUMBER;
BEGIN
FOR r IN c_merge LOOP
SELECT COUNT(*) INTO v_exists FROM O_RollCall WHERE Roll = r.Roll;

END LOOP;
EXCEPTION WHEN OTHERS THEN DBMS_OUTPUT.PUT_LINE('Error');
END;
/
